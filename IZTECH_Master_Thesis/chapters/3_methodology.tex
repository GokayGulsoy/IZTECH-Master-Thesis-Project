\chapter{METHODOLOGY}
\label{ch:methodology}

This chapter details the proposed Learnable Polynomial Approximation Network (LPAN) framework for achieving secure Transformer inference using Fully Homomorphic Encryption (FHE). We first describe the system architecture and polynomial approximation foundations, then present each methodological contribution in detail.

\section{System Architecture}
The privacy-preserving framework operates on a client-server model. The client encrypts the initial prompt using a public key before transmitting it to the cloud server. The server, hosting the LLM weights in plaintext, processes the encrypted embeddings using homomorphic addition and multiplication operations. The resulting encrypted output tokens are returned to the client, who decrypts them using the corresponding secret key.

\section{Polynomial Approximation Foundations}
\label{sec:poly_approx}

As established in Section~\ref{sec:nonlinearity_bottleneck}, FHE schemes natively support only addition and multiplication. Every non-polynomial function in the Transformer must be replaced by a polynomial of bounded degree. This section develops the mathematical foundations of the three principal approximation strategies.

\subsection{Taylor Series Approximation}

The Taylor series expansion of a function $f(x)$ around a point $a$ provides a natural polynomial approximation:
\begin{equation}
    f(x) = \sum_{k=0}^{\infty} \frac{f^{(k)}(a)}{k!}(x - a)^k,
    \label{eq:taylor_general}
\end{equation}
with the truncation error bounded by:
\begin{equation}
    \left| f(x) - T_n(x) \right| \leq \frac{M_{n+1}}{(n+1)!} |x - a|^{n+1},
    \label{eq:taylor_error}
\end{equation}
where $M_{n+1} = \max_{\xi \in [a, x]} |f^{(n+1)}(\xi)|$. Taylor approximation converges rapidly near $a$ but diverges at interval boundaries---problematic for FHE where inputs are not tightly controlled after several encrypted layers.

\subsection{Minimax (Chebyshev) Approximation}

Minimax approximation seeks the polynomial of degree~$n$ that minimizes the maximum absolute error over an entire interval $[a, b]$:
\begin{equation}
    p_n^* = \arg\min_{p \in \mathcal{P}_n} \max_{x \in [a, b]} |f(x) - p(x)|.
    \label{eq:minimax}
\end{equation}
The Chebyshev Equioscillation Theorem guarantees that $p_n^*$ is unique and characterised by at least $n+2$ equioscillation points where the error alternates in sign.

Minimax approximations are computed in the Chebyshev polynomial basis $\{T_k(x)\}_{k \geq 0}$, defined by the recurrence:
\begin{align}
    T_0(x) &= 1, \quad T_1(x) = x, \label{eq:cheb_base} \\
    T_{k+1}(x) &= 2x \cdot T_k(x) - T_{k-1}(x), \quad k \geq 1. \label{eq:cheb_recurrence}
\end{align}
For an arbitrary interval $[a, b]$, the affine map $\hat{x} = \frac{2x - (a+b)}{b - a}$ transforms to $[-1, 1]$. Chebyshev coefficients are evaluated at runtime using the numerically stable Clenshaw recurrence:
\begin{equation}
    f(x) \approx \sum_{k=0}^{n} c_k\, T_k(\hat{x}), \quad \text{evaluated via } b_k = c_k + 2\hat{x}\, b_{k+1} - b_{k+2}.
    \label{eq:clenshaw}
\end{equation}

\subsection{Least-Squares Regression Approximation}

Least-squares regression minimizes the mean squared error:
\begin{equation}
    p_n^{\text{LS}} = \arg\min_{p \in \mathcal{P}_n} \int_{a}^{b} \left[f(x) - p(x)\right]^2 w(x)\, dx.
    \label{eq:least_squares}
\end{equation}
When the weight function $w(x)$ matches the empirical activation distribution, least-squares can outperform minimax in average error, though at the cost of potentially larger worst-case error at boundaries.

\subsection{Approximation of the Inverse Square Root for LayerNorm}

LayerNorm requires computing $1/\sqrt{\sigma^2 + \epsilon}$. Two HE-compatible strategies exist:

\paragraph{Direct polynomial approximation.} Fit a low-degree polynomial to $g(x) = 1/\sqrt{x}$ over the expected variance range $[\epsilon, V_{\max}]$:
\begin{equation}
    \widetilde{g}(x) = \alpha_0 + \alpha_1 x + \alpha_2 x^2 + \cdots + \alpha_n x^n.
    \label{eq:invsqrt_poly}
\end{equation}

\paragraph{Goldschmidt iteration.} Iteratively refine an initial estimate $y_0 \approx 1/\sqrt{x}$:
\begin{align}
    h_i &= 1 - x \cdot y_i^2, \label{eq:goldschmidt_h} \\
    y_{i+1} &= y_i \cdot \left(1 + \frac{h_i}{2}\right). \label{eq:goldschmidt_y}
\end{align}
Each iteration consumes 3--4 multiplicative levels. Our LPAN framework uses the direct polynomial approach with learnable coefficients, avoiding the depth cost of iterative methods.

\subsection{Multiplicative Depth Analysis}
\label{sec:depth_analysis}

A degree-$n$ polynomial can be evaluated in $\lceil \log_2 n \rceil$ multiplicative depths using a balanced evaluation tree. For the complete secure Transformer layer:
\begin{equation}
    D_{\text{layer}} = D_{\mathrm{QKV}} + D_{\mathrm{softmax}} + D_{\mathrm{attn \cdot V}} + D_{\mathrm{LN}_1} + D_{\mathrm{FFN}_1} + D_{\mathrm{GELU}} + D_{\mathrm{FFN}_2} + D_{\mathrm{LN}_2}.
    \label{eq:layer_depth}
\end{equation}

\section{Distribution-Weighted Minimax Polynomial Initialisation}
\label{sec:weighted_minimax}

The standard minimax formulation treats all points in the approximation interval equally. However, during inference, inputs to non-linear functions follow an empirical distribution that depends on the model architecture and input data. This motivates a \emph{density-weighted} generalisation that focuses approximation effort on high-density regions.

\subsection{Problem Formulation}

Let $f: [a, b] \to \mathbb{R}$ be the target function and $\rho: [a, b] \to \mathbb{R}_{>0}$ a positive weight function derived from the empirical activation density. The \textbf{distribution-weighted minimax approximation} is:
\begin{equation}
    p_n^{**} = \arg\min_{p \in \mathcal{P}_n}\; \max_{x \in [a,b]}\; \rho(x) \cdot |f(x) - p(x)|.
    \label{eq:weighted_minimax}
\end{equation}

\subsection{Density Estimation via Kernel Density Estimation}

The weight function $\rho(x)$ is constructed from empirical activation samples collected by profiling the plaintext model. We attach forward hooks to the BERT model and record:
\begin{itemize}
    \item The inputs to each GELU activation (outputs of the intermediate dense layer),
    \item The scaled attention scores $QK^\top / \sqrt{d_k}$ before Softmax,
    \item The per-token variance $\sigma^2$ computed inside each LayerNorm.
\end{itemize}

From these samples, $\rho(x)$ is estimated using a Gaussian Kernel Density Estimator with Silverman's bandwidth:
\begin{equation}
    \hat{\rho}(x) = \frac{1}{nh} \sum_{i=1}^{n} K\!\left(\frac{x - x_i}{h}\right), \quad K(u) = \frac{1}{\sqrt{2\pi}} e^{-u^2/2},
    \label{eq:kde}
\end{equation}
where $h = 1.06\, \hat{\sigma}\, n^{-1/5}$ is Silverman's rule-of-thumb bandwidth. The 0.5th and 99.5th percentiles determine the effective approximation interval $[a, b]$, typically 30--50\% narrower than fixed intervals used in prior work.

\subsection{Theoretical Guarantees}

\begin{theorem}[Existence and Uniqueness of Weighted Minimax]
\label{thm:weighted_existence}
For any continuous function $f$ on $[a,b]$ and any continuous positive weight function $\rho > 0$ on $[a,b]$, there exists a unique polynomial $p_n^{**} \in \mathcal{P}_n$ solving~\eqref{eq:weighted_minimax}.
\end{theorem}

\begin{proof}
Define $g(x) = \rho(x) \cdot f(x)$ and the subspace $V_n = \{\rho(x) \cdot p(x) : p \in \mathcal{P}_n\}$. Since $\rho > 0$ on $[a,b]$, the map $p \mapsto \rho \cdot p$ is a linear isomorphism from $\mathcal{P}_n$ onto $V_n$. The problem is equivalent to $\min_{q \in V_n} \|g - q\|_\infty$, a standard best approximation from a Chebyshev subspace satisfying the Haar condition.
\end{proof}

\begin{theorem}[Error Bound for Weighted Minimax]
\label{thm:weighted_error}
Let $p_n^*$ be the standard minimax polynomial and $p_n^{**}$ the weighted minimax polynomial. Then:
\begin{equation}
    \|f - p_n^{**}\|_{\rho,\infty} \;\leq\; \|f - p_n^*\|_\infty \cdot \|\rho\|_\infty.
    \label{eq:weighted_error_bound}
\end{equation}
\end{theorem}

This guarantees the weighted solution is never worse than unweighted minimax on high-density regions.

\subsection{Role in the LPAN Framework}

The weighted minimax coefficients serve as the \emph{initialisation} for the learnable polynomial modules described in Section~\ref{sec:lpan}. Rather than starting from random coefficients or standard Chebyshev approximation, the distribution-weighted solution provides an informed starting point that is already adapted to the model's activation statistics. During subsequent training stages, these coefficients are further refined through backpropagation. As shown in Chapter~\ref{ch:results}, weighted minimax initialisation achieves $3.4\times$ lower weighted error than standard Chebyshev on GELU at degree~8.

\section{The LPAN Framework}
\label{sec:lpan}

The Learnable Polynomial Approximation Network (LPAN) is a three-stage progressive framework that transforms a pre-trained BERT model into a fully-polynomial variant where every non-linear operation is replaced by a learnable Chebyshev polynomial. This section describes the learnable polynomial modules, the progressive replacement strategy, and the knowledge distillation training procedure.

\subsection{Overview}

The LPAN pipeline consists of three stages, each targeting a different non-linear operation:
\begin{enumerate}
    \item \textbf{Stage 1 (GELU $\to$ Polynomial):} Replace all GELU activations simultaneously with learnable polynomial modules, fine-tune using cross-entropy loss.
    \item \textbf{Stage 2 (Softmax $\to$ Polynomial):} Replace Softmax layer-by-layer using progressive replacement with attention-level knowledge distillation.
    \item \textbf{Stage 3 (LayerNorm $\to$ Polynomial):} Replace LayerNorm layer-by-layer using progressive replacement with attention-level knowledge distillation and co-adaptation.
\end{enumerate}

Algorithm~\ref{alg:lpan} presents the complete LPAN pipeline.

\begin{algorithm}[htbp]
\caption{LPAN: Learnable Polynomial Approximation Network}
\label{alg:lpan}
\begin{algorithmic}[1]
\REQUIRE Pre-trained BERT model $\mathcal{M}$, training data $\mathcal{D}$, teacher model $\mathcal{T} \leftarrow \text{copy}(\mathcal{M})$
\ENSURE Fully-polynomial model $\mathcal{M}_{\text{poly}}$
\STATE \textbf{// Stage 1: GELU Replacement}
\STATE Profile activation distributions via forward hooks on $\mathcal{M}$
\STATE Compute weighted minimax coefficients for GELU at each layer
\STATE Replace all GELU activations with \textsc{PolynomialGELU} modules
\STATE Fine-tune $\mathcal{M}$ on $\mathcal{D}$ using $\mathcal{L}_{\mathrm{CE}}$ \COMMENT{All GELU simultaneously}
\STATE \textbf{// Stage 2: Progressive Softmax Replacement}
\FOR{$\ell = 0$ \TO $L-1$}
    \STATE Replace Softmax in layer $\ell$ with \textsc{PerHeadPolynomialSoftmax}
    \STATE Freeze all layers except layer $\ell$ and classifier
    \STATE Train using $\mathcal{L}_{\mathrm{AttnKD}}$ (Eq.~\ref{eq:attn_kd}) for $E_\ell$ epochs
\ENDFOR
\STATE \textbf{// Stage 3: Progressive LayerNorm Replacement}
\FOR{$\ell = 0$ \TO $L-1$}
    \STATE Replace both LayerNorms in layer $\ell$ with \textsc{PolynomialLayerNorm}
    \STATE Unfreeze layer $\ell$, all previously-replaced LN layers, and classifier \COMMENT{Co-adaptation}
    \STATE Train using $\mathcal{L}_{\mathrm{AttnKD}}$ (Eq.~\ref{eq:attn_kd}) for $E_\ell$ epochs
\ENDFOR
\RETURN $\mathcal{M}$ as $\mathcal{M}_{\text{poly}}$
\end{algorithmic}
\end{algorithm}

\subsection{Learnable Polynomial Modules}
\label{sec:poly_modules}

Each non-linear operation is replaced by a dedicated learnable module whose Chebyshev coefficients are \texttt{nn.Parameter} tensors optimised via backpropagation. All modules are initialised using the weighted minimax coefficients from Section~\ref{sec:weighted_minimax}.

\subsubsection{Polynomial GELU}

The GELU activation $\mathrm{GELU}(x) = x \cdot \Phi(x)$ is replaced by a degree-$d$ Chebyshev polynomial:
\begin{equation}
    \widetilde{\mathrm{GELU}}(x) = \sum_{k=0}^{d} c_k\, T_k\!\left(\frac{2x - (a+b)}{b-a}\right),
    \label{eq:poly_gelu}
\end{equation}
where $c_0, c_1, \ldots, c_d$ are learnable parameters, $[a, b]$ is the profiled approximation interval, and input values are clamped to $[a, b]$ during training. The polynomial degree $d$ is set adaptively based on layer depth: deeper layers use higher degrees to compensate for accumulated error.

\subsubsection{Per-Head Polynomial Softmax}
\label{sec:perhead_softmax}

Unlike prior approaches that share a single polynomial approximation of $e^x$ across all attention heads, LPAN assigns independent Chebyshev coefficients to each head:
\begin{equation}
    \widetilde{\exp}_h(x) = \sum_{k=0}^{d} c_{h,k}\, T_k\!\left(\frac{2x - (a+b)}{b-a}\right), \quad h = 1, \ldots, H,
    \label{eq:perhead_exp}
\end{equation}
where $H$ is the number of attention heads. The coefficient tensor has shape $[H, d+1]$, enabling each head to specialise its exponential approximation to the distinct score distributions it learns. The approximate Softmax is then computed as:
\begin{equation}
    \widetilde{\mathrm{softmax}}_h(x_i) = \frac{\widetilde{\exp}_h(x_i)}{\sum_{j} \widetilde{\exp}_h(x_j)}.
    \label{eq:perhead_softmax}
\end{equation}
Division is performed in plaintext during training; for encrypted inference, it is approximated by a polynomial reciprocal or Goldschmidt iteration.

The motivation for per-head independence is that different attention heads learn qualitatively different patterns---some attend broadly while others focus sharply---producing score distributions with different ranges and shapes. A shared polynomial must compromise between these patterns, while per-head coefficients can optimally approximate each head's specific distribution.

\subsubsection{Polynomial LayerNorm}
\label{sec:poly_layernorm}

LayerNorm (Equation~\ref{eq:layernorm}) contains the inverse square root $1/\sqrt{\sigma^2 + \epsilon}$, which we replace with a learnable polynomial:
\begin{equation}
    \widetilde{\mathrm{LN}}(x_i) = \gamma \cdot (x_i - \mu) \cdot \widetilde{g}(\sigma^2 + \epsilon) + \beta,
    \label{eq:poly_ln}
\end{equation}
where $\widetilde{g}(v) = \sum_{k=0}^{d} c_k T_k(\hat{v})$ approximates $1/\sqrt{v}$ and $\hat{v}$ is the affine map to $[-1, 1]$ over the profiled variance range. The mean $\mu$ and variance $\sigma^2$ are computed exactly (they involve only additions and multiplications, which are HE-native). The degree is set adaptively per layer, with deeper layers using higher degrees.

This is the first approach to replace LayerNorm's inverse square root with a learnable polynomial across all layers of a deep Transformer. Prior methods either skip LayerNorm (THE-X replaces it with an affine transformation $\hat{y} = x \circ \hat{\gamma} + \hat{\beta}$ that discards normalisation), delegate it to client-side computation, or handle it via interactive MPC protocols.


\subsection{Stage 1: GELU Replacement}
\label{sec:stage1}

In Stage~1, all GELU activations across all layers are replaced simultaneously with \textsc{PolynomialGELU} modules initialised from weighted minimax coefficients. The model is fine-tuned using standard cross-entropy loss on the downstream task:
\begin{equation}
    \mathcal{L}_{\mathrm{S1}} = \mathcal{L}_{\mathrm{CE}}(y, \hat{y}).
    \label{eq:stage1_loss}
\end{equation}

GELU can be replaced all-at-once (rather than layer-by-layer) because the feed-forward network is a local operation within each layer that does not propagate errors through the attention mechanism. The cross-entropy loss provides sufficient gradient signal for the polynomial coefficients to adapt.

\subsection{Stage 2: Progressive Softmax Replacement}
\label{sec:stage2}

Softmax replacement is more delicate than GELU because errors in attention weights directly affect the information flow across all token positions. LPAN therefore replaces Softmax \emph{one layer at a time}, proceeding from Layer~0 to Layer~$L-1$.

At each step $\ell$:
\begin{enumerate}
    \item The Softmax in layer $\ell$ is replaced with \textsc{PerHeadPolynomialSoftmax}.
    \item Only layer $\ell$ and the classification head are unfrozen.
    \item The model is trained for $E_\ell$ epochs using the attention-level knowledge distillation loss (Section~\ref{sec:attn_kd}).
\end{enumerate}

This progressive strategy ensures that each newly introduced polynomial has already stabilised before the next layer is modified, preventing the cascading error amplification that occurs with simultaneous replacement in deep models.

\subsection{Stage 3: Progressive LayerNorm Replacement}
\label{sec:stage3}

LayerNorm replacement is the most challenging stage because LayerNorm directly scales the hidden representations that feed into both attention and feed-forward sub-layers. A small error in the normalisation factor can shift the entire activation distribution, causing downstream polynomial approximations (GELU and Softmax, already replaced in Stages 1--2) to receive out-of-range inputs.

LPAN replaces LayerNorm layer-by-layer from Layer~0 to Layer~$L-1$, with a critical addition: \textbf{co-adaptation}. At each step $\ell$:
\begin{enumerate}
    \item Both LayerNorms in layer $\ell$ (post-attention and post-FFN) are replaced with \textsc{PolynomialLayerNorm}.
    \item Layer $\ell$, \emph{all previously-replaced LayerNorm layers} (layers $0, \ldots, \ell-1$), and the classification head are unfrozen.
    \item The model is trained using $\mathcal{L}_{\mathrm{AttnKD}}$ for $E_\ell$ epochs.
\end{enumerate}

Co-adaptation is essential because replacing a LayerNorm at layer $\ell$ shifts the activation statistics at that layer, which may require the polynomial LayerNorms at earlier layers to re-adjust. Without co-adaptation, accuracy degrades progressively and irrecoverably as more layers are replaced. With co-adaptation, the jointly refined ensemble of polynomial LayerNorms maintains stable activation distributions throughout the model.

For BERT-Base (12 layers), Stage~3 uses a variable epoch schedule: $E_\ell = 4$ for layers 0--3 (where initial stabilisation is critical), $E_\ell = 2$ for layers 4--9, and $E_\ell = 3$ for layers 10--11 (final layers), totalling 34 training epochs.


\subsection{Attention-Level Knowledge Distillation}
\label{sec:attn_kd}

Stages 2 and 3 use a composite knowledge distillation loss that supervises the student (polynomial) model at three levels:
\begin{equation}
    \mathcal{L}_{\mathrm{AttnKD}} = \alpha \cdot \mathcal{L}_{\mathrm{CE}}(y, \hat{y}) + \beta \cdot \sum_{\ell=1}^{L} \mathrm{KL}\!\left(A_\ell^{(\mathrm{T})} \,\|\, A_\ell^{(\mathrm{S})}\right) + \gamma \cdot \sum_{\ell=1}^{L} \mathrm{MSE}\!\left(H_\ell^{(\mathrm{T})}, H_\ell^{(\mathrm{S})}\right),
    \label{eq:attn_kd}
\end{equation}
where:
\begin{itemize}
    \item $\mathcal{L}_{\mathrm{CE}}$ is the task cross-entropy loss, ensuring the model retains task performance.
    \item $\mathrm{KL}(A^{(\mathrm{T})} \| A^{(\mathrm{S})})$ is the KL-divergence between teacher and student attention probability distributions at each layer, preserving the attention routing patterns.
    \item $\mathrm{MSE}(H^{(\mathrm{T})}, H^{(\mathrm{S})})$ is the mean squared error between teacher and student hidden states at each layer, preserving the intermediate representations.
    \item $\alpha = 1.0$, $\beta = 0.01$, $\gamma = 10.0$ are loss weighting hyperparameters.
\end{itemize}

The teacher model $\mathcal{T}$ is a frozen copy of the model state at the \emph{beginning of each stage}---not the original pre-trained model. This ensures the teacher's attention patterns and hidden states already account for the polynomial replacements made in previous stages.

\subsection{Learning Rate Strategy}
\label{sec:lr_strategy}

LPAN employs two learning rate techniques:

\paragraph{Layer-wise Learning Rate Decay (LLRD).} Different layers receive different base learning rates to account for their varying sensitivity:
\begin{equation}
    \mathrm{lr}_\ell = \mathrm{lr}_{\mathrm{base}} \cdot \left(1.0 + 0.5 \cdot \frac{\ell}{\max(1, L-1)}\right),
    \label{eq:llrd}
\end{equation}
where $\ell$ is the layer index. This gives deeper layers (which have larger gradients and require more adaptation) a higher learning rate, scaling from $1.0\times$ to $1.5\times$ the base rate.

\paragraph{Constant with warmup schedule.} Rather than a decaying schedule, Stages 2 and 3 use a constant learning rate with linear warmup over 10\% of training steps. This prevents the learning rate from decaying to ineffectiveness during the short per-layer training windows (2--4 epochs each).

\paragraph{Full-precision training.} Stage 3 (LayerNorm replacement) is trained in \texttt{fp32} rather than mixed-precision \texttt{fp16}, because the gradient magnitudes during LayerNorm polynomial training are extremely small (order $10^{-8}$) and cause the \texttt{fp16} gradient scaler to collapse. The base learning rate for Stage 3 is $5 \times 10^{-6}$.


\subsection{NaN-Safe Training}
\label{sec:nan_safe}

Polynomial approximations can produce extreme outputs when inputs fall outside the profiled interval, potentially causing NaN gradients that corrupt the entire parameter update. LPAN uses a NaN-safe training procedure:
\begin{enumerate}
    \item After loss computation, check for NaN/Inf in the loss value. If detected, skip the backward pass and log a warning.
    \item After gradient computation, scan all parameter gradients. Any individual gradient tensor containing NaN values is zeroed independently (rather than zeroing all gradients or skipping the entire step).
    \item Gradient clipping with max norm 1.0 is applied after NaN zeroing.
\end{enumerate}

This selective zeroing strategy is critical: zeroing only the corrupted gradients while preserving valid gradients from other parameters ensures training progress continues even when occasional inputs trigger polynomial overflow.

\section{Error Propagation Analysis}
\label{sec:error_propagation}

The previous sections optimise per-operation error. In a multi-layer Transformer, approximation errors compound through successive layers. This section derives formal bounds on the error amplification.

\subsection{Error Sources}

Three error sources contribute to the discrepancy between the original model output $\mathbf{y}$ and the polynomial-approximated output $\hat{\mathbf{y}}$:
\begin{enumerate}
    \item \textbf{Polynomial approximation error:} $\varepsilon_f(n) = \sup_{x \in I_f} |f(x) - p_n(x)|$ for each function $f$ approximated by degree-$n$ polynomial $p_n$.
    \item \textbf{CKKS scheme noise:} Each homomorphic multiplication introduces noise with accumulated standard deviation $O(\sqrt{m}\, \sigma_{\text{ckks}})$ after $m$ multiplications.
    \item \textbf{Compound propagation:} Errors from layer $\ell$ are amplified by the Lipschitz constants of operations in layers $\ell+1, \ldots, L$.
\end{enumerate}

\subsection{Per-Layer Error Bound}

\begin{theorem}[Layer Error Bound]
\label{thm:layer_error}
For a single Transformer layer with self-attention and feed-forward sub-layers, the output error is bounded by:
\begin{equation}
    \|\Delta \mathbf{h}^{(\ell)}\| \leq \underbrace{\sigma(W_O) \cdot n_{\text{seq}} \cdot \varepsilon_{\text{sm}} \cdot \sigma(W_V) \cdot \|\mathbf{h}\|}_{\text{Attention error}} + \underbrace{\sigma(W_2) \cdot \sqrt{d_{\text{ff}}} \cdot \varepsilon_{\text{gelu}}}_{\text{FFN error}} + \underbrace{2\, \varepsilon_{\text{ln}} \cdot \sqrt{d}}_{\text{LayerNorm error}},
    \label{eq:layer_error}
\end{equation}
where $\sigma(W)$ denotes the spectral norm of weight matrix $W$, $n_{\text{seq}}$ is the sequence length, $d_{\text{ff}}$ is the FFN intermediate dimension, and $d$ is the model hidden dimension.
\end{theorem}

\subsection{Multi-Layer Compound Error Bound}

\begin{theorem}[Compound Error Bound]
\label{thm:compound_error}
For an $L$-layer Transformer with residual connections, if each layer introduces error $e_\ell$ and the amplification factor per residual block is $\alpha = 1 + \max_\ell \sigma(W_O^{(\ell)}, W_2^{(\ell)})$, then:
\begin{equation}
    \|\Delta \mathbf{h}^{(L)}\| \leq \sum_{\ell=0}^{L-1} \alpha^{L-1-\ell} \cdot e_\ell.
    \label{eq:compound_error}
\end{equation}
For uniform per-layer error $e_\ell = e$:
\begin{equation}
    \|\Delta \mathbf{h}^{(L)}\| \leq e \cdot \frac{\alpha^L - 1}{\alpha - 1}.
    \label{eq:compound_uniform}
\end{equation}
\end{theorem}

\begin{corollary}[Depth Requirement]
\label{cor:depth_requirement}
To ensure total error below $\varepsilon_{\text{target}}$, the per-layer error must satisfy:
\begin{equation}
    e \leq \varepsilon_{\text{target}} \cdot \frac{\alpha - 1}{\alpha^L - 1},
    \label{eq:depth_requirement}
\end{equation}
which, for BERT-Base ($L=12$, $\alpha \approx 6.16$), requires $e < 4.6 \times 10^{-10} \cdot \varepsilon_{\text{target}}$. This exponential tightening motivates the progressive replacement strategy: rather than fighting against this error amplification with ever-higher polynomial degrees, LPAN uses knowledge distillation to ensure each layer's output closely matches the teacher, effectively keeping $\alpha$ close to 1 during training.
\end{corollary}

\section{Polynomial Approximations for Non-Linearity}
\label{sec:poly_approx}

As established in Section~\ref{sec:nonlinearity_bottleneck}, a fundamental limitation of FHE is the inability to compute non-linear functions directly. Every non-polynomial function in the Transformer---Softmax, GELU, and LayerNorm---must be replaced by a polynomial of bounded degree that can be evaluated using only homomorphic additions and multiplications. The choice of approximation method, the polynomial degree, and the approximation interval collectively determine both the accuracy of the secure inference and the multiplicative depth consumed. This section develops the mathematical foundations of the three principal approximation strategies employed in this work: Taylor series expansion, Minimax (Chebyshev) approximation, and least-squares regression fitting.

\subsection{Taylor Series Approximation}

The Taylor series expansion of a function $f(x)$ around a point $a$ provides a natural polynomial approximation:
\begin{equation}
    f(x) = \sum_{k=0}^{\infty} \frac{f^{(k)}(a)}{k!}(x - a)^k = f(a) + f'(a)(x-a) + \frac{f''(a)}{2!}(x-a)^2 + \cdots
    \label{eq:taylor_general}
\end{equation}
Truncating this series at degree $n$ yields the Taylor polynomial $T_n(x)$, with the approximation error bounded by the Lagrange remainder:
\begin{equation}
    \left| f(x) - T_n(x) \right| \leq \frac{M_{n+1}}{(n+1)!} |x - a|^{n+1},
    \label{eq:taylor_error}
\end{equation}
where $M_{n+1} = \max_{\xi \in [a, x]} |f^{(n+1)}(\xi)|$.

\subsubsection{Taylor Approximation of GELU}

Expanding the GELU function (Equation~\ref{eq:gelu}) around $a = 0$ yields:
\begin{equation}
    \mathrm{GELU}(x) \approx 0.5x + \frac{x}{\sqrt{2\pi}} - \frac{x^3}{6\sqrt{2\pi}} + \cdots
    \label{eq:gelu_taylor}
\end{equation}
A commonly used degree-2 approximation is~\cite{lee2022lowcomplexity}:
\begin{equation}
    \widetilde{\mathrm{GELU}}_2(x) = 0.5x + 0.398x^2,
    \label{eq:gelu_deg2}
\end{equation}
which consumes only one multiplicative level (for the $x^2$ term) but introduces significant error for $|x| > 2$. A degree-4 approximation offers substantially better fidelity:
\begin{equation}
    \widetilde{\mathrm{GELU}}_4(x) = 0.5x + 0.398x - 0.0215x^3 + 0.000217x^4,
    \label{eq:gelu_deg4}
\end{equation}
consuming two multiplicative levels when evaluated via the factored form $x^4 = (x^2)^2$.

\paragraph{Limitations.} The Taylor approximation converges rapidly near the expansion point $a$ but can diverge badly at the boundaries of the approximation interval. For FHE, where inputs are not tightly controlled (especially after several encrypted layers), this boundary divergence can cause catastrophic accuracy loss. Moreover, the Taylor series does not minimize the \emph{worst-case} error over an interval---it only ensures local accuracy near~$a$.

\subsubsection{Taylor Approximation of Softmax}

Direct Taylor expansion of the full Softmax function is impractical because it couples all elements of the input vector through the normalizing denominator. Instead, the standard approach approximates the exponential component individually. The degree-2 Taylor expansion of $e^x$ around $a = 0$ is:
\begin{equation}
    e^x \approx 1 + x + \frac{x^2}{2}.
    \label{eq:exp_taylor2}
\end{equation}

Substituting this into the Softmax definition gives an approximate numerator and denominator that can each be computed homomorphically, though the final division must itself be approximated (e.g., via Goldschmidt iteration or a polynomial approximation of $1/x$)~\cite{hao2022iron}.

\subsection{Minimax (Chebyshev) Approximation}

While Taylor polynomials are anchored to a single expansion point, \emph{Minimax approximation} seeks the polynomial of degree~$n$ that minimizes the maximum absolute error over an entire interval $[a, b]$:
\begin{equation}
    p_n^* = \arg\min_{p \in \mathcal{P}_n} \max_{x \in [a, b]} |f(x) - p(x)|,
    \label{eq:minimax}
\end{equation}
where $\mathcal{P}_n$ denotes the set of all polynomials of degree at most~$n$. The existence and uniqueness of $p_n^*$ is guaranteed by the Chebyshev Equioscillation Theorem:

\begin{definition}[Chebyshev Equioscillation Theorem]
A polynomial $p_n^* \in \mathcal{P}_n$ is the unique best approximation to $f$ on $[a, b]$ in the $L^\infty$ norm if and only if the error $f(x) - p_n^*(x)$ equioscillates---that is, there exist at least $n + 2$ points $a \leq x_0 < x_1 < \cdots < x_{n+1} \leq b$ such that
\begin{equation}
    f(x_i) - p_n^*(x_i) = (-1)^i \lambda, \quad \lambda = \pm\|f - p_n^*\|_\infty.
    \label{eq:equioscillation}
\end{equation}
\end{definition}

The practical construction of minimax polynomials uses the Remez algorithm, which iteratively refines the equioscillation points until convergence. For FHE applications, the minimax approach is superior to Taylor expansion because it distributes the approximation error uniformly across the interval rather than concentrating accuracy near a single point.

\subsubsection{Chebyshev Polynomial Basis}

Minimax approximations are typically computed in the \emph{Chebyshev polynomial basis} $\{T_k(x)\}_{k \geq 0}$, defined on $[-1, 1]$ by the recurrence:
\begin{align}
    T_0(x) &= 1, \quad T_1(x) = x, \label{eq:cheb_base} \\
    T_{k+1}(x) &= 2x \cdot T_k(x) - T_{k-1}(x), \quad k \geq 1. \label{eq:cheb_recurrence}
\end{align}
Equivalently, $T_k(x) = \cos(k \cdot \arccos(x))$. Any continuous function $f$ on $[-1, 1]$ can be expanded as:
\begin{equation}
    f(x) \approx \sum_{k=0}^{n} c_k\, T_k(x), \quad c_k = \frac{2}{\pi} \int_{-1}^{1} \frac{f(x)\, T_k(x)}{\sqrt{1 - x^2}}\, dx,
    \label{eq:cheb_expansion}
\end{equation}
where the prime on the sum indicates that the $k = 0$ term is halved. For an arbitrary interval $[a, b]$, the change of variable $\hat{x} = \frac{2x - (a+b)}{b - a}$ maps to $[-1, 1]$.

Truncating the Chebyshev expansion at degree $n$ yields a near-minimax polynomial whose maximum error is within a factor of $\mathcal{O}(\log n)$ of the optimal minimax error~\cite{trefethen2019approximation}. This near-optimality, combined with the numerical stability of the Chebyshev basis (which avoids the ill-conditioning of the monomial basis at high degrees), makes it the standard tool for constructing FHE-compatible activation function approximations.

\subsubsection{Minimax Approximation of GELU}

Over the interval $[-5, 5]$ (which covers the typical activation range observed during LLM inference), the degree-4 minimax polynomial for GELU is:
\begin{equation}
    \widetilde{\mathrm{GELU}}_4^{\text{mm}}(x) \approx -0.0048x^4 + 0.1025x^3 + 0.4738x^2 + 0.1594x - 0.0672,
    \label{eq:gelu_minimax4}
\end{equation}
with a maximum absolute error of approximately $5.7 \times 10^{-2}$ over $[-5, 5]$---substantially better than the Taylor degree-4 approximation over the same interval, which exhibits errors exceeding $0.5$ near the boundaries.

\subsubsection{Minimax Approximation of Softmax}

For the exponential kernel $e^x$ over a bounded interval $[-B, B]$ (where $B$ is determined by the expected range of attention logits), a degree-$n$ minimax polynomial can be computed via the Remez algorithm. For $n = 4$ on $[-8, 0]$ (the numerically stable range after subtracting the row maximum), the minimax approximation achieves a relative error below $10^{-2}$, which is sufficient for attention weight computation since Softmax outputs are subsequently multiplied by value vectors~\cite{hao2022iron}.

\subsection{Least-Squares Regression Approximation}

An alternative to minimax optimization is \emph{least-squares regression}, which minimizes the mean squared error over a set of sample points or a continuous distribution:
\begin{equation}
    p_n^{\text{LS}} = \arg\min_{p \in \mathcal{P}_n} \int_{a}^{b} \left[f(x) - p(x)\right]^2 w(x)\, dx,
    \label{eq:least_squares}
\end{equation}
where $w(x) \geq 0$ is a weight function. When the weight function matches the distribution of actual activations observed during inference, least-squares approximation can outperform minimax in practice: it achieves lower \emph{average} error at the cost of potentially larger worst-case error at the interval boundaries.

For FHE applications, least-squares approximation is particularly useful when the input distribution is known from profiling the plaintext model. By fitting the polynomial to the empirical activation distribution rather than optimizing for the worst case, one can achieve better end-to-end model accuracy with lower-degree polynomials~\cite{lee2022lowcomplexity}.

\subsection{Approximation of the Inverse Square Root for LayerNorm}

The LayerNorm operation (Equation~\ref{eq:layernorm}) requires computing $1/\sqrt{\sigma^2 + \epsilon}$. Two HE-compatible strategies exist:

\paragraph{Direct polynomial approximation.} Fit a low-degree polynomial to $g(x) = 1/\sqrt{x}$ over the expected variance range $[\epsilon, V_{\max}]$:
\begin{equation}
    \widetilde{g}(x) = \alpha_0 + \alpha_1 x + \alpha_2 x^2 + \cdots + \alpha_n x^n.
    \label{eq:invsqrt_poly}
\end{equation}
This requires prior knowledge of the variance range, which can be obtained by profiling the plaintext model.

\paragraph{Goldschmidt iteration.} Iteratively refine an initial estimate $y_0 \approx 1/\sqrt{x}$ using:
\begin{align}
    h_i &= 1 - x \cdot y_i^2, \label{eq:goldschmidt_h} \\
    y_{i+1} &= y_i \cdot \left(1 + \frac{h_i}{2}\right). \label{eq:goldschmidt_y}
\end{align}
Each iteration consumes approximately 3--4 multiplicative levels but quadratically improves precision. Two iterations typically suffice to achieve the accuracy required for neural network inference~\cite{hao2022iron}.

\subsection{Multiplicative Depth Analysis}
\label{sec:depth_analysis}

A degree-$n$ polynomial can be evaluated in $\lceil \log_2 n \rceil$ multiplicative depths using a balanced evaluation tree. For example, a degree-4 polynomial $p(x) = a_0 + a_1 x + a_2 x^2 + a_3 x^3 + a_4 x^4$ is computed as:
\begin{align}
    \text{Level 1:}& \quad x^2 = x \cdot x, \label{eq:eval_l1} \\
    \text{Level 2:}& \quad x^4 = x^2 \cdot x^2, \quad x^3 = x^2 \cdot x, \label{eq:eval_l2}
\end{align}
followed by scalar multiplications (which are ``free'' in depth, as plaintext-ciphertext multiplications do not increase ciphertext degree) and additions to form the final result. Each level requires a subsequent rescaling operation per Section~\ref{sec:rescaling}, consuming one modulus level.

For the complete secure Transformer layer, the multiplicative depth budget is allocated as follows:
\begin{equation}
    D_{\text{layer}} = D_{\mathrm{QKV}} + D_{\mathrm{softmax}} + D_{\mathrm{attn \cdot V}} + D_{\mathrm{LN}_1} + D_{\mathrm{FFN}_1} + D_{\mathrm{GELU}} + D_{\mathrm{FFN}_2} + D_{\mathrm{LN}_2},
    \label{eq:layer_depth}
\end{equation}
where $D_{\mathrm{QKV}}$ and $D_{\mathrm{FFN}_{1,2}}$ each consume one level for the matrix-vector multiplication, and the non-linear terms consume $\lceil \log_2 n_i \rceil$ levels each. A typical configuration with degree-2 Softmax, degree-4 GELU, and degree-2 inverse-square-root yields $D_{\text{layer}} \approx 4 + 2 + 1 + 2 + 1 + 2 + 1 + 2 = 15$ levels per Transformer layer. The selection of specific polynomial degrees and the resulting accuracy--depth trade-offs are evaluated experimentally in Chapter~4.

\section{Approximation-Aware Fine-Tuning Strategy}
\label{sec:approx_finetuning}

A critical insight from recent secure inference literature---particularly THE-X~\cite{chen2022thex}---is that the polynomial approximation and the model fine-tuning should not be treated as independent steps. Naively replacing non-linear functions with polynomial approximations in a pre-trained model causes \emph{compounding approximation errors} that propagate through successive layers, leading to unacceptable accuracy degradation. Instead, the model must be \emph{re-trained} (fine-tuned) with the approximations already in place, allowing the network parameters to compensate for the approximation-induced perturbations.

This section describes the approximation-aware fine-tuning strategy adopted in our framework, drawing on the staged workflow proposed by THE-X and extending it with the polynomial approximation techniques from Section~\ref{sec:poly_approx}.

\subsection{Staged Approximation Workflow}

Our framework follows a multi-stage approximation process:

\paragraph{Stage 1: Activation replacement and standard fine-tuning.}
The GELU activation in the feed-forward network is replaced with its polynomial approximation $\widetilde{\mathrm{GELU}}_n(x)$ (Equations~\ref{eq:gelu_deg2}--\ref{eq:gelu_minimax4}). Similarly, the Softmax function in the attention mechanism is replaced with its polynomial or rational approximation. The model is then fine-tuned on the downstream task with these substitutions in place, using the standard cross-entropy or regression loss. This allows the linear projection weights ($W_Q, W_K, W_V, W_1, W_2$) to adapt to the approximated activation landscape.

\paragraph{Stage 2: LayerNorm approximation via distillation.}
After Stage~1 converges, the original LayerNorm layers are replaced with their polynomial or affine approximations (Section~\ref{sec:depth_analysis}). The approximated LayerNorm parameters are then optimized by minimizing the MSE loss between the output of the approximated layer and the output of the original LayerNorm, following the knowledge distillation approach:
\begin{equation}
    \mathcal{L}_{\text{LN}} = \sum_{k=1}^{K} \left\| O_k - \widetilde{O}_k \right\|_2^2,
    \label{eq:ln_distill}
\end{equation}
where $O_k$ and $\widetilde{O}_k$ are the outputs of the $k$-th original and approximated LayerNorm, respectively. All parameters except those of the approximated LayerNorm are frozen during this stage.

\paragraph{Rationale for staged optimization.}
THE-X~\cite{chen2022thex} empirically demonstrated that jointly optimizing all approximation components simultaneously leads to inferior results compared to the staged approach. The authors attributed this to a bi-level optimization difficulty: when both the Softmax approximation and the LayerNorm approximation are trained jointly, their error surfaces interact in complex ways that prevent convergence to a good solution. Our framework adopts the same principle, optimizing each non-linear replacement in isolation before composing them.

\subsection{Handling Numerical Stability in Encrypted Inference}
\label{sec:numerical_stability}

Several numerical challenges arise specifically in the encrypted domain that do not occur in plaintext inference:

\paragraph{Attention mask values.}
Standard Transformer implementations use negative infinity ($-\infty$) to mask out padding tokens before Softmax, ensuring zero attention weight for padding positions. However, polynomial approximations of $e^x$ diverge catastrophically for large negative inputs. Following THE-X's empirical finding~\cite{chen2022thex}, masked positions should use a finite negative value (typically between $-3$ and $-5$) rather than $-\infty$. This ensures the polynomial approximation remains in its valid range while still effectively suppressing attention to padding tokens.

\paragraph{Attention score overflow.}
Before LayerNorm, the multi-head attention output can exhibit extreme values (up to $10^4$) that fall far outside the polynomial approximation interval. Weight decay regularization during fine-tuning has been shown to effectively constrain the attention score magnitudes~\cite{chen2022thex}. In our framework, we additionally profile the activation ranges of each layer during plaintext inference to select appropriate approximation intervals $[a, b]$ for each non-linear function.

\paragraph{Precision management.}
Each homomorphic multiplication followed by rescaling introduces a small rounding error. Over the course of a deep circuit ($D_{\text{total}} = 15 \times N_{\text{layers}}$ levels), these rounding errors accumulate and can erode the precision of intermediate computations. The CKKS scaling factor $\Delta$ must be chosen large enough to maintain sufficient bits of precision throughout the circuit. As shown in the CKKS paper~\cite{cheon2017homomorphic}, the precision loss during evaluation is bounded by the depth of the circuit and exceeds at most one additional bit compared to unencrypted floating-point arithmetic (see Section~\ref{sec:rescaling}).

\subsection{Our Approach Versus THE-X}
\label{sec:our_vs_thex}

While our framework adopts the staged approximation workflow from THE-X, it differs in several important respects:

\begin{enumerate}
    \item \textbf{Pure polynomial approach.} THE-X replaces GELU with ReLU and delegates the $\max(0, x)$ computation to the client, introducing additional communication rounds. Our framework instead approximates GELU directly with Chebyshev/minimax polynomials, eliminating the need for client-server interaction during inference. This yields a \emph{non-interactive} secure inference protocol.

    \item \textbf{Mathematically grounded Softmax.} THE-X uses a learned neural network to estimate Softmax, which provides no formal error guarantees. Our approach uses polynomial approximations of the exponential function combined with polynomial approximation of the reciprocal (or Goldschmidt iteration), enabling provable error bounds on the Softmax approximation.

    \item \textbf{LayerNorm treatment.} THE-X simplifies LayerNorm to a pure affine transformation, discarding the normalization effect entirely. Our framework retains the normalization structure by approximating the inverse square root $1/\sqrt{\sigma^2 + \epsilon}$ as a low-degree polynomial, preserving the stabilizing effect of normalization while remaining HE-compatible.

    \item \textbf{Depth budget optimization.} By using optimized polynomial degrees (selected via the minimax criterion), our framework aims to minimize the total multiplicative depth while maintaining accuracy, rather than offloading operations to the client.
\end{enumerate}

\section{Proposed Contribution 1: Distribution-Weighted Minimax Polynomial Approximation}
\label{sec:weighted_minimax}

The standard minimax formulation in Equation~\eqref{eq:minimax} treats all points in the approximation interval $[a, b]$ equally. However, during actual inference, the inputs to non-linear functions are not uniformly distributed---they follow an empirical distribution that depends on both the model architecture and the input data. This observation motivates a \emph{density-weighted} generalisation of minimax approximation that focuses approximation accuracy on the regions of highest activation density.

\subsection{Problem Formulation}

Let $f: [a, b] \to \mathbb{R}$ be the target non-linear function (GELU, $\exp$, or $1/\sqrt{x}$), and let $\rho: [a, b] \to \mathbb{R}_{>0}$ be a positive weight function derived from the empirical activation density. We define the \textbf{distribution-weighted minimax approximation} as:
\begin{equation}
    p_n^{**} = \arg\min_{p \in \mathcal{P}_n}\; \max_{x \in [a,b]}\; \rho(x) \cdot |f(x) - p(x)|.
    \label{eq:weighted_minimax}
\end{equation}

This is equivalent to finding the best uniform approximation to the modified function $\rho(x) \cdot f(x)$ within the space of functions $\{\rho(x) \cdot p(x) : p \in \mathcal{P}_n\}$, which relates to classical weighted Chebyshev approximation theory~\cite{trefethen2019approximation}.

\subsection{Density Estimation via Kernel Density Estimation}

The weight function $\rho(x)$ is constructed from empirical activation samples collected by profiling the plaintext model on representative task data. Specifically, we attach forward hooks to BERT-Tiny and record:
\begin{itemize}
    \item The inputs to each GELU activation (outputs of the intermediate dense layer),
    \item The scaled attention scores $QK^\top / \sqrt{d_k}$ before Softmax,
    \item The per-token variance $\sigma^2$ computed inside each LayerNorm.
\end{itemize}

From these samples, we estimate $\rho(x)$ using a Gaussian Kernel Density Estimator (KDE) with Silverman's bandwidth selection~\cite{silverman1986density}:
\begin{equation}
    \hat{\rho}(x) = \frac{1}{nh} \sum_{i=1}^{n} K\!\left(\frac{x - x_i}{h}\right), \quad K(u) = \frac{1}{\sqrt{2\pi}} e^{-u^2/2},
    \label{eq:kde}
\end{equation}
where $h = 1.06\, \hat{\sigma}\, n^{-1/5}$ is Silverman's rule-of-thumb bandwidth.

The 0.5th and 99.5th percentiles of the empirical samples determine the effective approximation interval $[a, b]$, which is typically 30--50\% narrower than the fixed intervals used in prior work (e.g., $[-5, 5]$ for GELU).

\subsection{Theoretical Guarantees}

The weighted minimax formulation inherits the strong theoretical properties of classical Chebyshev approximation:

\begin{theorem}[Existence and Uniqueness of Weighted Minimax]
\label{thm:weighted_existence}
For any continuous function $f$ on $[a,b]$ and any continuous positive weight function $\rho > 0$ on $[a,b]$, there exists a unique polynomial $p_n^{**} \in \mathcal{P}_n$ solving the weighted minimax problem~\eqref{eq:weighted_minimax}.
\end{theorem}

\begin{proof}
Define $g(x) = \rho(x) \cdot f(x)$ and the subspace $V_n = \{\rho(x) \cdot p(x) : p \in \mathcal{P}_n\}$. Since $\rho > 0$ on $[a,b]$, the map $p \mapsto \rho \cdot p$ is a linear isomorphism from $\mathcal{P}_n$ onto $V_n$. The problem~\eqref{eq:weighted_minimax} is equivalent to:
\[
    \min_{q \in V_n} \|g - q\|_\infty,
\]
which is a standard best approximation problem in the Chebyshev norm from the $(n+1)$-dimensional Chebyshev subspace $V_n$. Since $V_n$ satisfies the Haar condition (as the product of a positive function with a Chebyshev system remains a Chebyshev system), the Chebyshev Equioscillation Theorem guarantees existence and uniqueness.
\end{proof}

\begin{theorem}[Error Bound for Weighted Minimax]
\label{thm:weighted_error}
Let $p_n^*$ be the standard (unweighted) minimax polynomial for $f$ on $[a,b]$, and let $p_n^{**}$ be the weighted minimax polynomial with weight $\rho$. Then:
\begin{equation}
    \|f - p_n^{**}\|_{\rho,\infty} \;\leq\; \|f - p_n^*\|_\infty \cdot \|\rho\|_\infty,
    \label{eq:weighted_error_bound}
\end{equation}
where $\|\cdot\|_{\rho,\infty}$ denotes the weighted supremum norm $\max_{x \in [a,b]} \rho(x)|f(x) - p(x)|$.
\end{theorem}

\begin{proof}
Since $p_n^{**}$ minimises $\max_{x} \rho(x)|f(x)-p(x)|$ over all $p \in \mathcal{P}_n$, taking $p = p_n^*$ as a feasible (but not necessarily optimal) solution gives:
\[
    \|f - p_n^{**}\|_{\rho,\infty} \leq \max_{x \in [a,b]} \rho(x) |f(x) - p_n^*(x)| \leq \|\rho\|_\infty \cdot \|f - p_n^*\|_\infty.
\]
\end{proof}

This result shows that the weighted solution is \emph{never worse} than the unweighted minimax on high-density regions; it can only improve by redistributing approximation effort away from low-density tails.

\subsection{Computational Method}

In practice, we solve~\eqref{eq:weighted_minimax} using a modified Remez exchange algorithm. First, we transform the problem to the standard Chebyshev basis on $[-1, 1]$ via the affine map $\hat{x} = \frac{2x - (a+b)}{b-a}$. Then, the weighted Remez iteration:
\begin{enumerate}
    \item Initialises with $n+2$ Chebyshev reference points.
    \item Solves the linear system for the equioscillation condition with weight $\rho$.
    \item Locates the new extremum of $\rho(x)|f(x) - p(x)|$ and exchanges the worst reference point.
    \item Repeats until the oscillation magnitudes converge within tolerance $\epsilon = 10^{-12}$.
\end{enumerate}

The resulting Chebyshev coefficients $\{c_k\}_{k=0}^{n}$ are evaluated at runtime using the numerically stable Clenshaw recurrence:
\begin{equation}
    f(x) \approx \sum_{k=0}^{n} c_k\, T_k(\hat{x}), \quad \text{evaluated via } b_k = c_k + 2\hat{x}\, b_{k+1} - b_{k+2}.
    \label{eq:clenshaw}
\end{equation}

\section{Proposed Contribution 2: Layer-Adaptive CKKS Depth Allocation}
\label{sec:depth_allocation}

All existing CKKS-based secure Transformer systems assign the \emph{same} polynomial degree to every instance of each non-linear function, regardless of the layer. This section formulates the depth allocation problem as a constrained integer programme and presents an efficient dynamic programming solution.

\subsection{Problem Formulation}

Consider a Transformer with $L$ layers, where each layer contains three non-linear operations indexed by $i \in \mathcal{I} = \{\text{GELU}, \text{Softmax}, \text{LN}\}$. Let $n_i^{(\ell)} \in \{2, 4, 6, 8\}$ denote the polynomial degree assigned to operation $i$ at layer $\ell$. The CKKS multiplicative depth consumed by a degree-$n$ polynomial evaluated via balanced binary tree is $d(n) = \lceil \log_2 n \rceil$.

The depth allocation problem is:
\begin{equation}
    \min_{\mathbf{n} \in \mathbb{Z}_{+}^{|\mathcal{I}| \cdot L}} \;\;
    \sum_{\ell=1}^{L} \sum_{i \in \mathcal{I}}
    \mathcal{E}\!\left(n_i^{(\ell)},\, \rho_i^{(\ell)}\right)
    \quad \text{subject to} \quad
    \sum_{\ell=1}^{L} \sum_{i \in \mathcal{I}} d\!\left(n_i^{(\ell)}\right) \leq D_{\text{budget}},
    \label{eq:depth_allocation}
\end{equation}
where $\mathcal{E}(n, \rho)$ is the weighted minimax approximation error (Equation~\ref{eq:weighted_minimax}) for degree $n$ under the empirical density $\rho_i^{(\ell)}$ profiled at layer $\ell$, and $D_{\text{budget}}$ is the total multiplicative depth available from the CKKS parameter set.

\subsection{Dynamic Programming Solution}

Since the candidate degrees are drawn from a small set $\{2, 4, 6, 8\}$ and the number of operations is $|\mathcal{I}| \cdot L$ (for BERT-Tiny, $3 \times 2 = 6$), the problem decomposes as a bounded knapsack:

\begin{theorem}[Optimal Substructure]
\label{thm:dp_optimal}
Let $\text{OPT}(k, D)$ denote the minimum total error achievable using operations $1, \ldots, k$ with depth budget $D$. Then:
\begin{equation}
    \text{OPT}(k, D) = \min_{n \in \{2,4,6,8\}:\; d(n) \leq D}
    \left[\mathcal{E}_k(n) + \text{OPT}(k-1,\, D - d(n))\right],
    \label{eq:dp_recurrence}
\end{equation}
with base case $\text{OPT}(0, D) = 0$ for all $D \geq 0$.
\end{theorem}

The DP table is computed in $O(|\mathcal{I}| \cdot L \cdot |\mathcal{D}| \cdot D_{\text{budget}})$ time, where $|\mathcal{D}|$ is the number of candidate degrees. For BERT-Tiny with $L=2$, $|\mathcal{I}|=3$, $|\mathcal{D}|=4$, and $D_{\text{budget}} \leq 30$, this requires fewer than $1000$ operations.

\subsection{Intuition and Key Insight}

The fundamental insight is that different layers exhibit different activation distributions, leading to vastly different error-vs-degree trade-offs. A layer whose activations concentrate in a narrow range (e.g., GELU inputs at Layer~0 with $\sigma = 1.19$) can be well approximated by a low-degree polynomial, while a layer with a wide, heavy-tailed distribution (e.g., Softmax inputs at Layer~1 with range $[-7.2, 8.4]$) requires a higher degree. Uniform allocation wastes depth budget on layers that do not need it, while starving layers that are accuracy-critical.

\subsection{Pipeline Summary}

The complete optimisation pipeline combines Contributions~1 and~2:
\begin{enumerate}
    \item \textbf{Profile activations:} Run BERT-Tiny on SST-2 training data with forward hooks recording GELU inputs, Softmax scores, and LayerNorm variances at each layer.
    \item \textbf{Estimate densities:} Fit KDE to each activation type at each layer, determining the effective interval $[a_i^{(\ell)}, b_i^{(\ell)}]$ from the 0.5th--99.5th percentiles.
    \item \textbf{Build error table:} For each $(i, \ell)$ and each candidate degree $n \in \{2, 4, 6, 8\}$, compute $\mathcal{E}(n, \rho_i^{(\ell)})$ via the weighted minimax solver.
    \item \textbf{Solve depth allocation:} Apply the DP recurrence~\eqref{eq:dp_recurrence} to obtain the globally optimal degree assignment $\mathbf{n}^*$.
    \item \textbf{Approximation-aware fine-tuning:} Replace GELU, Softmax, and LayerNorm with polynomial modules using the optimal coefficients, then fine-tune the model end-to-end on the downstream task.
    \item \textbf{CKKS encrypted inference:} Deploy the fine-tuned model with the polynomial activations under CKKS encryption.
\end{enumerate}

\section{Proposed Contribution 3: Evolutionary Optimisation of Polynomial Parameters}
\label{sec:ga_optimization}

While the dynamic programming approach of Contribution~2 finds the globally optimal \emph{degree assignment} for a fixed set of approximation intervals, it treats three critical design decisions independently: (i)~the polynomial degree, (ii)~the approximation interval width, and (iii)~the density weighting function. In practice, these parameters interact non-linearly---widening an interval at one layer may permit a lower degree at the next layer by reducing the range of downstream activations. This section formulates a joint optimisation over all three axes using a genetic algorithm (GA).

\subsection{Search Space and Chromosome Encoding}

For a Transformer with $L$~layers and $|\mathcal{I}|$~non-linear operations per layer, the search space has $K = L \cdot |\mathcal{I}|$ genes, each encoding two decision variables:
\begin{itemize}
    \item \textbf{Degree gene:} $n_k \in \{2, 4, 6, 8\}$ --- a discrete degree choice.
    \item \textbf{Interval gene:} $\alpha_k \in [0, 1]$ --- a continuous expansion factor that widens the $k$-th profiled interval $[a_k, b_k]$ by a factor of $\alpha_k \cdot \gamma_{\max}$ on each side, where $\gamma_{\max} = 0.5$ bounds the maximum expansion to 50\% of the base range.
\end{itemize}
The expanded interval for operation $k$ is:
\begin{equation}
    [\tilde{a}_k, \tilde{b}_k] = \left[a_k - \alpha_k \gamma_{\max} (b_k - a_k),\;\; b_k + \alpha_k \gamma_{\max} (b_k - a_k)\right].
    \label{eq:ga_interval}
\end{equation}
The total chromosome length is $2K$ (for BERT-Tiny, $K = 6$, giving 12 genes).

\subsection{Fitness Function}

Unlike the DP formulation, which minimises per-operation approximation error, the GA directly optimises a \emph{fidelity-based} fitness function that measures how well the polynomial-replaced model preserves the original model's output distribution:
\begin{equation}
    \text{fitness}(\mathbf{c}) = \frac{1}{1 + D_{\mathrm{KL}}\!\left(P_{\text{base}} \;\|\; P_{\text{poly}}(\mathbf{c})\right)} - \lambda \cdot \max\!\left(0,\, \sum_{k=1}^{K} d(n_k) - D_{\text{budget}}\right),
    \label{eq:ga_fitness}
\end{equation}
where $P_{\text{base}}$ and $P_{\text{poly}}(\mathbf{c})$ are the softmax output distributions of the baseline and polynomial-replaced models on a validation subset, $D_{\mathrm{KL}}$ denotes the Kullback--Leibler divergence, and $\lambda > 0$ is the depth-budget penalty coefficient.

The key advantage of this formulation is that it captures \emph{inter-layer error interactions}: a configuration that introduces moderate error at Layer~0 but keeps Layer~1 outputs stable may achieve higher fidelity than one that minimises Layer~0 error in isolation but causes downstream amplification.

\subsection{Genetic Operators}

The GA employs:
\begin{itemize}
    \item \textbf{Tournament selection} with tournament size $k = 3$, balancing selection pressure and diversity.
    \item \textbf{Two-point crossover} applied jointly to both degree and interval genes, with crossover probability $p_c = 0.8$.
    \item \textbf{Hybrid mutation} ($p_m = 0.15$ per gene):
    \begin{itemize}
        \item Degree mutation: move to an adjacent degree in the ordered set $\{2, 4, 6, 8\}$.
        \item Interval mutation: Gaussian perturbation $\alpha_k \leftarrow \text{clip}(\alpha_k + \mathcal{N}(0, 0.15),\; 0,\; 1)$.
    \end{itemize}
    \item \textbf{Elitism}: the top 2 individuals are carried forward unchanged.
\end{itemize}

\subsection{Convergence and Complexity}

The GA evaluates $P \cdot G$ fitness function calls, where $P$ is the population size and $G$ the number of generations. Each evaluation requires a forward pass of the polynomial-replaced model on $N_{\text{eval}}$ samples. For BERT-Tiny with $P = 20$, $G = 25$, and $N_{\text{eval}} = 200$, the total cost is approximately $500 \times 200 = 100{,}000$ forward-pass samples. This is substantially cheaper than grid search over $|\mathcal{D}|^K = 4^6 = 4096$ degree combinations (each requiring $N_{\text{eval}}$ samples), and the GA additionally searches the continuous interval space.

\section{Proposed Contribution 4: Depth-Optimal Polynomial Evaluation Strategies}
\label{sec:bsgs_eval}

Once the polynomial coefficients and degrees have been determined, the evaluation strategy---the order in which multiplications are performed---determines the \emph{multiplicative depth} consumed. This section analyses three strategies: Horner's method, balanced binary tree evaluation, and the Paterson--Stockmeyer (baby-step giant-step) algorithm.

\subsection{Horner's Method}

The classical Horner evaluation for a degree-$n$ polynomial $p(x) = \sum_{k=0}^{n} c_k x^k$:
\begin{equation}
    p(x) = c_0 + x\bigl(c_1 + x(c_2 + \cdots + x\,c_n)\bigr)
    \label{eq:horner}
\end{equation}
requires exactly $n$ multiplications, all sequential, yielding multiplicative depth $d_{\text{Horner}}(n) = n$. For degree-8 polynomials, this consumes 8 of the available levels---prohibitively expensive under tight CKKS budgets.

\subsection{Balanced Binary Tree Evaluation}

The balanced tree approach pre-computes powers of $x$ by repeated squaring:
\begin{equation}
    x^1 \to x^2 = x \cdot x \to x^4 = x^2 \cdot x^2 \to x^8 = x^4 \cdot x^4 \to \cdots
    \label{eq:balanced_tree}
\end{equation}
The final polynomial value is assembled as:
\begin{equation}
    p(x) = \underbrace{(c_0 + c_1 x + c_2 x^2 + c_3 x^3)}_{\text{low half}} + x^4 \cdot \underbrace{(c_4 + c_5 x + c_6 x^2 + c_7 x^3)}_{\text{high half}},
    \label{eq:balanced_eval}
\end{equation}
where each half is itself computed recursively. The multiplicative depth is:
\begin{equation}
    d_{\text{tree}}(n) = \lceil \log_2 n \rceil.
    \label{eq:tree_depth}
\end{equation}
For degree 8, this gives $d_{\text{tree}}(8) = 3$, a $2.7\times$ reduction over Horner. However, the number of ciphertext--ciphertext multiplications is higher ($2n - 1$ vs.\ $n$).

\subsection{Paterson--Stockmeyer Algorithm}

The Paterson--Stockmeyer (P-S) algorithm~\cite{paterson1973number} achieves a balanced trade-off. For a degree-$n$ polynomial, it sets step size $s = \lceil\sqrt{n}\rceil$ and decomposes:
\begin{equation}
    p(x) = \sum_{j=0}^{\lceil n/s \rceil - 1} \left(\sum_{i=0}^{s-1} c_{js+i}\, x^i\right) \cdot x^{js},
    \label{eq:ps_decomposition}
\end{equation}
where the ``baby-step'' powers $\{1, x, x^2, \ldots, x^{s-1}\}$ are computed first, then the ``giant-step'' powers $\{1, x^s, x^{2s}, \ldots\}$ are formed by repeated squaring of $x^s$. The multiplicative depth is:
\begin{equation}
    d_{\text{PS}}(n) = \lceil \log_2 s \rceil + \lceil \log_2 \lceil n/s \rceil \rceil \approx 2\lceil \log_2 \sqrt{n} \rceil.
    \label{eq:ps_depth}
\end{equation}

\begin{theorem}[P-S Complexity]
\label{thm:ps_complexity}
The Paterson--Stockmeyer algorithm evaluates a degree-$n$ polynomial using at most $2\sqrt{n}$ ciphertext--ciphertext multiplications and multiplicative depth $O(\log n)$.
\end{theorem}

\section{Proposed Contribution 5: Error Propagation Bounds for Polynomial-Approximated Transformers}
\label{sec:error_propagation}

The previous contributions optimise per-operation error. However, in a multi-layer Transformer, approximation errors \emph{compound}: error introduced at Layer~$\ell$ propagates through all subsequent layers. This section derives formal bounds on how various error sources interact.

\subsection{Error Sources}

Three error sources contribute to the discrepancy between the original model output $\mathbf{y}$ and the encrypted polynomial-approximated output $\hat{\mathbf{y}}$:
\begin{enumerate}
    \item \textbf{Polynomial approximation error:} $\varepsilon_f(n) = \sup_{x \in I_f} |f(x) - p_n(x)|$ for each non-linear function $f$ approximated by degree-$n$ polynomial $p_n$ on interval $I_f$.
    \item \textbf{CKKS scheme noise:} Each homomorphic multiplication introduces noise with variance $\sigma_{\text{ckks}}^2$. After $m$ multiplications, the accumulated noise has standard deviation $O(\sqrt{m}\, \sigma_{\text{ckks}})$.
    \item \textbf{Compound propagation:} Errors from layer $\ell$ are amplified by the Lipschitz constants of operations in layers $\ell+1, \ldots, L$.
\end{enumerate}

\subsection{Per-Layer Error Bound}

\begin{theorem}[Layer Error Bound]
\label{thm:layer_error}
For a single Transformer layer with self-attention and feed-forward sub-layers, the output error is bounded by:
\begin{equation}
    \|\Delta \mathbf{h}^{(\ell)}\| \leq \underbrace{\sigma(W_O) \cdot n_{\text{seq}} \cdot \varepsilon_{\text{sm}} \cdot \sigma(W_V) \cdot \|\mathbf{h}\|}_{\text{Attention error}} + \underbrace{\sigma(W_2) \cdot \sqrt{d_{\text{ff}}} \cdot \varepsilon_{\text{gelu}}}_{\text{FFN error}} + \underbrace{2\, \varepsilon_{\text{ln}} \cdot \sqrt{d}}_{\text{LayerNorm error}},
    \label{eq:layer_error}
\end{equation}
where $\sigma(W)$ denotes the spectral norm of weight matrix $W$, $n_{\text{seq}}$ is the sequence length, $d_{\text{ff}}$ is the FFN intermediate dimension, and $d$ is the model hidden dimension.
\end{theorem}

\begin{proof}
Consider the attention sub-layer. The softmax approximation error $\varepsilon_{\text{sm}}$ affects each of the $n_{\text{seq}}$ scores per query position, producing an attention weight error bounded by $n_{\text{seq}} \cdot \varepsilon_{\text{sm}}$ (since the softmax partition function $Z \geq 1$). This error propagates through the value projection $W_V$ and output projection $W_O$, scaling by their respective spectral norms.

For the FFN sub-layer, $\text{FFN}(\mathbf{x}) = W_2 \cdot \text{GELU}(W_1 \mathbf{x} + \mathbf{b}_1) + \mathbf{b}_2$. The GELU approximation introduces per-element error $\varepsilon_{\text{gelu}}$ across $d_{\text{ff}}$ intermediate activations. The $\ell_2$ norm of this error vector is $\sqrt{d_{\text{ff}}} \cdot \varepsilon_{\text{gelu}}$, which is then scaled by $\sigma(W_2)$.

The LayerNorm error $\varepsilon_{\text{ln}}$ enters through both post-attention and post-FFN normalisations, contributing $2 \varepsilon_{\text{ln}} \sqrt{d}$.
\end{proof}

\subsection{Multi-Layer Compound Error Bound}

\begin{theorem}[Compound Error Bound]
\label{thm:compound_error}
For an $L$-layer Transformer with residual connections, if each layer $\ell$ introduces error $e_\ell$ (as bounded in Theorem~\ref{thm:layer_error}), and the amplification factor per residual block is $\alpha = 1 + \max_\ell \sigma(W_O^{(\ell)}, W_2^{(\ell)})$, then the total output error after $L$ layers satisfies:
\begin{equation}
    \|\Delta \mathbf{h}^{(L)}\| \leq \sum_{\ell=0}^{L-1} \alpha^{L-1-\ell} \cdot e_\ell.
    \label{eq:compound_error}
\end{equation}
For uniform per-layer error $e_\ell = e$:
\begin{equation}
    \|\Delta \mathbf{h}^{(L)}\| \leq e \cdot \frac{\alpha^L - 1}{\alpha - 1}.
    \label{eq:compound_uniform}
\end{equation}
\end{theorem}

\begin{proof}
Due to the residual connection, $\mathbf{h}^{(\ell+1)} = \mathbf{h}^{(\ell)} + f_\ell(\mathbf{h}^{(\ell)})$. The error at layer $\ell+1$ satisfies:
\[
    \|\Delta \mathbf{h}^{(\ell+1)}\| = \|\Delta \mathbf{h}^{(\ell)} + \Delta f_\ell(\mathbf{h}^{(\ell)})\| \leq \|\Delta \mathbf{h}^{(\ell)}\| + L_f \cdot \|\Delta \mathbf{h}^{(\ell)}\| + e_\ell = \alpha \|\Delta \mathbf{h}^{(\ell)}\| + e_\ell,
\]
where $L_f$ is the Lipschitz constant of the sub-layer function. Unrolling this recurrence yields~\eqref{eq:compound_error}.
\end{proof}

\begin{corollary}[Depth Requirement]
\label{cor:depth_requirement}
To ensure the total model error does not exceed a target $\varepsilon_{\text{target}}$, the per-operation polynomial degree must satisfy:
\begin{equation}
    e \leq \varepsilon_{\text{target}} \cdot \frac{\alpha - 1}{\alpha^L - 1},
    \label{eq:depth_requirement}
\end{equation}
which, combined with the known convergence rate of Chebyshev approximation ($\varepsilon_f(n) = O(\rho^{-n})$ for analytic $f$ with Bernstein ellipse parameter $\rho > 1$), determines the minimum degree required at each layer.
\end{corollary}
